# image annotation을 위한 구글 클라우드 비전 모듈 불러오기 
client = vision.ImageAnnotatorClient()

# chrome driver 접속
driver = webdriver.Chrome('./chromedriver')

# 코디 개수를 셀 카운터 변수
count = 1
# 해시태그 저장을 위한 딕셔너리 변수
dic = {}

# 각 의류 label 분류
top = ['Top']
bottom = ['Skirt', 'Miniskirt', 'Pants', 'Shorts']
shoe = ['Shoe', 'Boot']

for user in tqdm(celeb):
    for page in range(1, 12):
        # celeb 한 명의 개인 페이지(한 페이지 씩) 접속
        driver.get('https://codibook.net/user/' + user + '?&page=' + str(page))
        for i in range(2, 26):
            # 한 페이지에 속하는 코디 링크들을 하나씩 접속
            string = 'div#content > div:nth-child(' + str(i) + ') > div > div > div.thumb_wrapper > a'
            time.sleep(0.5)
            btn = driver.find_elements_by_tag_name(string)
            if not btn:
                break
            btn[0].click()

            # 한 개의 코디 페이지 소스 가져오기
            html = driver.page_source
            soup = BeautifulSoup(html, 'html.parser')

            # 해시태그 담을 변수
            tag = []

            # 해시태그 담아서 딕셔너리에 저장
            for i in soup.select('div.content > div.description > a'):
                tag.append(i.text[1:])
            dic[count] = tag

            # 이미지 담고 있는 부분 찾기
            div = soup.select_one('div#content_wrapper')

            # 로컬 드라이브 내 한 개 코디 이미지들 저장할 폴더 만들기
            os.mkdir('./data/img/codi_' + str(count))

            check = []
            try:
                # 코디 내 아이템 이미지 링크들 하나씩 있는 리스트
                for url in div.select('div.thumb_wrapper > img'):
                    # 이미지 링크 바로 binary 형태로 불러오기
                    content = urllib.request.urlopen(url['src']).read()
                    # Google Cloud Vision 모듈로 이미지 전달하여 라벨 확인하기
                    image = vision.Image(content=content)
                    annot = client.object_localization(image=image).localized_object_annotations
                    # 라벨 확인이 안되는 경우 NaN으로 기록
                    if len(annot) == 0:
                        label = 'NaN'
                    else:
                        label = annot[0].name

                    clothes = ''
                    check.append(label)
                    if label in bottom:
                        clothes = '하의'
                    elif label in top:
                        clothes = '상의'
                    elif label in shoe:
                        clothes = '신발'
                    else:
                        continue
                    urllib.request.urlretrieve(url['src'], './data/img/codi_' + str(count) + '/' + clothes + '.png')

                if len(os.listdir('./data/img/codi_' + str(count))) != 3:
                    shutil.rmtree(r'./data/img/codi_' + str(count) + '/')
                    print(count,'번째: ', check)

                count += 1
                driver.back()
            except:
                print('Nonetype')
